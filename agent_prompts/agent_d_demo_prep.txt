RESET DEMO SCRIPT:

### reset_demo.sh
```bash
#!/bin/bash
# Reset demo database to clean state

echo "Resetting demo database..."

# Backup current DB
if [ -f "quiz_warehouse.db" ]; then
    cp quiz_warehouse.db "quiz_warehouse_backup_$(date +%Y%m%d_%H%M%S).db"
    echo "Backed up current database"
fi

# Create fresh DB
rm -f quiz_warehouse.db
python -c "from src.migrations import initialize_db; initialize_db('quiz_warehouse.db')"

echo "Demo database reset complete"
echo "Run demo setup:"
echo "  python demo_data/setup_demo.py"
```

### demo_data/setup_demo.py
```python
"""
Setup demo data for workshop presentation.
Run this before starting the demo.
"""
import sys
sys.path.insert(0, '..')

from src.database import get_engine, get_session
from src.classroom import create_class
from src.lesson_tracker import log_lesson

def setup_demo():
    engine = get_engine('quiz_warehouse.db')
    session = get_session(engine)

    # Create demo class
    demo_class = create_class(
        session,
        name="7th Grade Science - Block A",
        grade_level="7th Grade",
        subject="Science",
        standards=["SOL 7.1", "SOL 7.2", "SOL 7.3"]
    )

    print(f"Created class: {demo_class.name} (ID: {demo_class.id})")

    # Log demo lessons
    log_lesson(
        session,
        class_id=demo_class.id,
        content="Taught photosynthesis: light-dependent and light-independent reactions.",
        topics=["photosynthesis", "chloroplasts", "light energy"],
        notes="Class engaged, good questions"
    )

    log_lesson(
        session,
        class_id=demo_class.id,
        content="Reviewed cellular respiration and compared to photosynthesis.",
        topics=["cellular respiration", "mitochondria", "ATP"],
        notes="Need to reinforce aerobic vs anaerobic"
    )

    print("Demo data created successfully!")
    session.close()

if __name__ == "__main__":
    setup_demo()
```
